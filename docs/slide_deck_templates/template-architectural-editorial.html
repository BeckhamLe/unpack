Under the Hood

The AI scheduling engine adapts intervals based on real-time recall patterns

core/scheduler.ts 
interface ReviewSignal {
  cardId: string;
  responseTime: number;     // ms to answer
  confidence: 1 | 2 | 3 | 4;  // self-reported
}

async function computeNextReview(
  signal: ReviewSignal,
  history: ReviewSignal[]
): Promise<Date> {
  const decay = estimateDecayRate(history);
  const boost = signal.confidence >= 3 ? 1.8 : 0.6;
  const interval = decay * boost * difficultyWeight(signal);

  return addHours(new Date(), interval);
}
